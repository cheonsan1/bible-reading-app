구글 Apps Script(code.gs) 연동 코드

이 코드를 구글 스프레드시트의 [확장 프로그램] -> [Apps Script] 창에 붙여넣어 배포하시면 실시간 자동 연동이 활성화됩니다.

/**
 * 천산중앙교회 성경통독 25주 통합 관리 양방향 시트 연동 스크립트
 */

function doPost(e) {
  var params = JSON.parse(e.postData.contents);
  var action = params.action;
  
  if (action === 'progress') {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
    var memberName = params.memberName;
    var completedDays = params.completedDays;
    
    var data = sheet.getDataRange().getValues();
    var targetRow = -1;
    for (var i = 1; i < data.length; i++) {
      if (data[i][1] == memberName) { 
        targetRow = i + 1;
        break;
      }
    }
    
    if (targetRow == -1) {
      sheet.appendRow(["W1", memberName, params.group, completedDays.length]);
      targetRow = sheet.getLastRow();
    }
    
    sheet.getRange(targetRow, 4).setValue(completedDays.length);
    
    for (var d = 1; d <= 150; d++) {
      var colIndex = d + 4; 
      if (completedDays.indexOf(d) > -1) {
        sheet.getRange(targetRow, colIndex).setValue("완독");
      } else {
        sheet.getRange(targetRow, colIndex).setValue("");
      }
    }
  } 
  else if (action === 'transcription') {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var secondSheetName = "필사 및 묵상 나눔";
    var memoSheet = ss.getSheetByName(secondSheetName);
    
    if (!memoSheet) {
      memoSheet = ss.insertSheet(secondSheetName);
      memoSheet.appendRow(["기입 시각", "성도명", "소속 조", "성경 범위", "묵상 내용", "필사노트 사진 주소"]);
    }
    
    var timeStamp = new Date();
    var memberName = params.memberName;
    var group = params.group;
    var verse = params.verse;
    var content = params.content;
    var imageUrl = params.imageUrl; 
    
    memoSheet.appendRow([timeStamp, memberName, group, verse, content, imageUrl]);
  }
  
  return ContentService.createTextOutput(JSON.stringify({result: "success"}))
    .setMimeType(ContentService.MimeType.JSON);
}
